#define EXCEPTION_STACK_SIZE 72

/* ========================================================= [ entry ] === */
  .section .text
reset_handler:
 
  /* MSTATUS:0x00001800,  MTVEC  :0x00000011*/
  addi x1, x0, 0x180  
  slli x1, x1, 4
  csrrw  x1, mstatus, x1 
  csrrwi x1, mtvec, 0x11
  
  # by ali
  # Enable machine-mode software interrupts (MSIE), timer interrupts (MTIE), and external interrupts (MEIE)
  li t0, 0x888             # Load immediate for enabling MSIE (0x8), MTIE (0x80), and MEIE (0x800)
  csrs mie, t0             # Set these bits in the mie register
  # Enable global interrupts by setting the MIE (Machine Interrupt Enable) bit in mstatus
  li t0, 0x8               # Load immediate: 0x8 enables global machine interrupts (MIE bit in mstatus)
  csrs mstatus, t0         # Set the MIE bit in the mstatus register

  
  /* set all registers to zero */
  mv  x1, x0
  mv  x2, x1
  mv  x3, x1
  mv  x4, x1
  mv  x5, x1
  mv  x6, x1
  mv  x7, x1
  mv  x8, x1
  mv  x9, x1
  mv x10, x1
  mv x11, x1
  mv x12, x1
  mv x13, x1
  mv x14, x1
  mv x15, x1
  mv x16, x1
  mv x17, x1
  mv x18, x1
  mv x19, x1
  mv x20, x1
  mv x21, x1
  mv x22, x1
  mv x23, x1
  mv x24, x1
  mv x25, x1
  mv x26, x1
  mv x27, x1
  mv x28, x1
  mv x29, x1
  mv x30, x1
  mv x31, x1

  /* stack initilization */
  la   x2, _stack_start


_start:
  .global _start

  /* clear BSS */
  la x26, _bss_start
  la x27, _bss_end

  bge x26, x27, zero_loop_end

zero_loop:
  sw x0, 0(x26)
  addi x26, x26, 4
  ble x26, x27, zero_loop
zero_loop_end:

main_entry:
  /* jump to main program entry point (argc = argv = 0) */
  addi x10, x0, 0
  addi x11, x0, 0
  /* enable all external interrupt, modified Luca */
  addi x7, x0, 0
  lui x7, 0xffff0
  csrrs x7, mie, x7
  addi x7, x0, 8
  csrrs x7, mstatus, x7
  addi x7, x0, 0

  jal x1, main

_end:
  /* when returning from main, infinite loop here */
  jal x0, _end


// saves all caller-saved registers (except return address)
store_regs:
  sw  x3, 0x00(x2)  // gp
  sw  x4, 0x04(x2)  // tp
  sw  x5, 0x08(x2)  // t0
  sw  x6, 0x0c(x2)  // t1
  sw  x7, 0x10(x2)  // t2
  sw x10, 0x14(x2)  // a0
  sw x11, 0x18(x2)  // a1
  sw x12, 0x1c(x2)  // a2
  sw x13, 0x20(x2)  // a3
  sw x14, 0x24(x2)  // a4
  sw x15, 0x28(x2)  // a5
  sw x16, 0x2c(x2)  // a6
  sw x17, 0x30(x2)  // a7
  sw x28, 0x34(x2)  // t3
  sw x29, 0x38(x2)  // t4
  sw x30, 0x3c(x2)  // t5
  sw x31, 0x40(x2)  // t6
  jalr x0, x1

// load back registers from stack
end_except:
  lw  x3, 0x00(x2)
  lw  x4, 0x04(x2)
  lw  x5, 0x08(x2)
  lw  x6, 0x0c(x2)
  lw  x7, 0x10(x2)
  lw x10, 0x14(x2)
  lw x11, 0x18(x2)
  lw x12, 0x1c(x2)
  lw x13, 0x20(x2)
  lw x14, 0x24(x2)
  lw x15, 0x28(x2)
  lw x16, 0x2c(x2)
  lw x17, 0x30(x2)
  lw x28, 0x34(x2)
  lw x29, 0x38(x2)
  lw x30, 0x3c(x2)
  lw x31, 0x40(x2)
  lw  x1, 0x44(x2)
  addi x2, x2, EXCEPTION_STACK_SIZE
  // j default_exc_handler
  // return back to main program, modified Luca
  mret

/* ecall trap will continue the main program in user mode, trap code to be implemented */

ecall_u_handler:
  addi x2, x2, -EXCEPTION_STACK_SIZE
  sw x1, 0x44(x2)
  jal x1, store_regs
  
  /* mepc+4 allows mret to continue the program after a ecall, added Luca */
  csrrsi x7, mepc, 0
  addi x7,x7,4
  csrrw x7,mepc,x7
  
  j end_except

external_interrupt_signal_handler_0:
  # Allocate exception stack
  addi x2, x2, -4
  # Save RA
  sw   x1, 0x44(x2)
  # ==== Enable hardware temporary registers ====
  # Use x1 to load the constant safely (allowed before switch)
  li   x1, 0x600
  csrs mstatus, x1      # Enable HW temp registers (bits 9+10)
  # --------------------------------------------------------------------------
  # FROM THIS POINT ON: t0, t1, t2, ... etc are SAFE to use (hardware bank)
  # --------------------------------------------------------------------------
  # Use temp registers now safely
  # Load address for hardware control
  lui  t0, 0xFFFF0
  addi t0, t0, 0x0C8
  # Write value 1
  li   t1, 1
  sw   t1, 0(t0)
  # Call your C/ASM handler
  jal  x1, EXT_INT_0_handler
  # ==== Disable hardware temporary registers ====
  li   t0, 0x400       # mask = bit 10
  csrc mstatus, t0     # clear bit 10
  # Clear mip
  csrw mip, x0
  # Restore RA (original bank now active)
  lw   x1, 0x44(x2)
  # Restore stack
  addi x2, x2, 4
  mret

  jal  x1, EXT_INT_15_handler

  li   t0, 0x400       # mask = bit 10

  # Clear mip
  csrw mip, x0
  # Restore RA (original bank now active)
  lw   x1, 0x44(x2)
  # Restore stack
  addi x2, x2, 4
